subseqCpDNA () {
	local subgen=$1
	local geneid=$2
	local outfas=$3
	tmp_list="tmp.lst"
	if [[ $subgen == "SS" ]]; then
		all_fa=$cpfa_ss
	elif [[ $subgen == "St" ]]; then
		all_fa=$cpfa_st
	elif [[ $subgen == "BB" ]]; then
		all_fa=$cpfa_bb
	else
		echo "Error: unknown genome: $all_fa=$cpfa_st" >&2
		return 100
	fi
	echo "    Test: fasta: $all_fa"
	seqarr=($(grep ^'>' $all_fa | grep -w "$geneid" | perl -lne 's/\s+.*//;s/^>//;print;'))
	if [ ${#seqarr[@]} -eq 0 ]; then
		echo "Error: empty seq: $geneid" >&2
		echo "    grep ^'>' $all_fa | grep -w \"$geneid\" | perl -lne 's/\s+.*//;s/^>//;print;'" >&2
		return 100
	elif [ ${#seqarr[@]} -eq 1 ]; then
		echo ">$subgen.$geneid" >> $outfas
		samtools faidx $all_fa "${seqarr[0]}" | grep -v ^'>' >> $outfas
	elif [ ${#seqarr[@]} -gt 1 ]; then
		for ((i=0;i<${#seqarr[@]};i++)); do
			j=$((i+1))
			echo ">$subgen.$geneid.$j" >> $outfas
			samtools faidx $all_fa "${seqarr[$i]}" | grep -v ^'>' >> $outfas
		done
	fi
	return 0
}
