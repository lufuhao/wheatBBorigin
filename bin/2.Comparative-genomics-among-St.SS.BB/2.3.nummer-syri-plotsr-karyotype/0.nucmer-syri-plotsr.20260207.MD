delta2syri () {
	infile=$1
	inref=$2
	inqry=$3
	outfile=$4
	if [ ! -s "$infile.coords" ]; then
		echo "### show-coords"
		show-coords -THrd $infile > $infile.coords
		if [ $? -ne 0 ] || [ ! -s "$infile.coords" ]; then
			echo "Error: show-coords failed" >&2
			return 100
		fi
	fi
	if [ ! -s "$outfile" ]; then
		echo "### syri"
		syri --nosnp -c $infile.coords -d $infile -r $inref -q $inqry --prefix ${outfile%syri.out}
		if [ $? -ne 0 ] || [ ! -s "$outfile" ]; then
			echo "Error: syri failed" >&2
			return 100
		fi
	fi
	return 0
}

genomeInfo () {
	GIfile=$1
	GIchr=$2
	cat > $GIfile <<EOGI
##ft	=   File type (fa/cl for fasta/chromosome_length, default = fa); cl files must be in tsv format with chromosome name in column 1 and chromosome length in column 2; using cl files is much faster than using fasta files
##lc	=	line colour
##lw	=	line width
#file	name	tags
SS${GIchr}.chrlen	SS	ft:cl;lw:3;lc:black
BB${GIchr}.chrlen	BB	ft:cl;lw:3;lc:black
St${GIchr}.chrlen	St	ft:cl;lw:3;lc:black
EOGI
	return 0
}
genomeCfg () {
	GCfile=$1
	cat > $GCfile <<EOCFG
## COLOURS and transparency for alignments (syntenic, inverted, translocated, and duplicated)
syncol:#008000
invcol:#FFA500
tracol:#9ACD32
dupcol:#00BBFF
alpha:1

## Margins and dimensions:
## Adjusts the gap between chromosomes and tracks. Higher values leads to more gap
chrmar:0.1
## Extra margin at the top and bottom of plot area
exmar:0.1

## LEGEND
## To plot legend use T, use F to not plot legend
legend:F
## Number of columns for genome legend, set -1 for automatic setup
genlegcol:-1
## [Left edge, bottom edge, width, height]
bbox:0.3,1.01,0.5,0.3
## For vertical chromosomes (using -v option)
##bbox_v:0,1.1,0.5,0.3
## Margin between genome and annotation legends
bboxmar:0.3
EOCFG
	return 0
}
